##### 源代码

```c
#include <stdio.h>

int main() {

    const double pi = 3.14159;
    double r, h;
    printf("input r and h: ");
    scanf("%lf %lf", &r, &h);

    printf("s = %.4lf\n", 2 * pi * r * h);
    printf("ssum = %.4lf\n", 2 * pi * r * h + 2 * pi * r * r);
    printf("v = %.4lf\n", pi * r * r * h);

    return 0;
}
```

## Ex13

##### 源代码

```c
#include <stdio.h>

void bin(int n) {
    int b[8] = {0};
    for (int i = 0; i < 8; i++) {
        if (n) {
            b[7 - i] = n % 2;
            n /= 2;
        }
        else
            break;
    }

    for (int i = 0; i < 8; i++) {
        printf("%2d", b[i]);
    }
}

int main() {

    int n;
    
    for (int i = 0; i < 8; i++) {
        printf("input a number(< 256): ");
        scanf("%d", &n);
        if (n > 255 || n < 0)
            break;
        printf("%d ---> ", n);

        bin(n);
        printf("\n");
    }

    return 0;
}
```
